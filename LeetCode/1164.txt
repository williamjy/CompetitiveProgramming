SELECT product_id, new_price AS price FROM Products WHERE (product_id, change_date) IN (SELECT P2.product_id, MAX(P2.change_date) AS change_date FROM Products P2 WHERE DATEDIFF(P2.change_date, '2019-08-16') <= 0 GROUP BY P2.product_id) UNION SELECT product_id, 10 AS price FROM Products GROUP BY product_id HAVING DATEDIFF(MIN(change_date), '2019-08-16') > 0;
